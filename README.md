# hyprtaskbarWindows
Windows like taskbar for hyprland

# Required packages
- aylurs-gtk-shell-git
- libastal-tray-git
- libastal-notifd-git
- libastal-wireplumber-git
- libastal-hyprland-git
- libastal-network-git


# How to run
- ags run ./app.ts

## Run with debug enabled
- GTK_DEBUG=interactive ags run ./app.ts
